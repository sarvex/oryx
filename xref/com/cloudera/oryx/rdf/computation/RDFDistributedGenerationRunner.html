72    private static final Logger log = LoggerFactory.getLogger(RDFDistributedGenerationRunner.class);
73  
74    @Override
75    protected List<DependsOn<Class<? extends JobStep>>> getPreDependencies() {
76      return Collections.emptyList();
77    }
78  
79    @Override
80    protected List<DependsOn<Class<? extends JobStep>>> getIterationDependencies() {
81      return Collections.singletonList(
82          DependsOn.<Class<? extends JobStep>>nextAfterFirst(BuildTreesStep.class, MergeNewOldStep.class));
83    }
84  
85    @Override
86    protected List<DependsOn<Class<? extends JobStep>>> getPostDependencies() {
87      return Collections.emptyList();
88    }
89  
90    @Override
91    protected JobStepConfig buildConfig(int iteration) {
92      return new RDFJobStepConfig(getInstanceDir(),
93                                  getGenerationID(),
94                                  getLastGenerationID(),
95                                  iteration);
96    }

98    @Override
99    protected void doPost() throws IOException {
100 
101     String instanceGenerationPrefix =
102         Namespaces.getInstanceGenerationPrefix(getInstanceDir(), getGenerationID());
103     String outputPathKey = instanceGenerationPrefix + "trees/";
104     Store store = Store.get();
105     PMML joinedPMML = null;
106 
107     // TODO This is still loading all trees into memory, which can be quite large.
108     // To do better we would have to manage XML output more directly.
109 
110     Map<String,Mean> columnNameToMeanImportance = Maps.newHashMap();
111 
112     for (String treePrefix : store.list(outputPathKey, true)) {
113       log.info("Reading trees from file {}", treePrefix);
114       for (String treePMMLAsLine : new FileLineIterable(store.readFrom(treePrefix))) {
115         PMML nextPMML;
116         try {
117           nextPMML = JAXBUtil.unmarshalPMML(
118               ImportFilter.apply(new InputSource(new StringReader(treePMMLAsLine))));
119         } catch (JAXBException e) {
120           throw new IOException(e);
121         } catch (SAXException e) {
122           throw new IOException(e);
123         }
124 
125         updateMeanImportances(columnNameToMeanImportance,
126                               nextPMML.getModels().get(0).getMiningSchema().getMiningFields());
127 
128         if (joinedPMML == null) {
129 
130           // No prior model. Just use this one as the current model.
131           joinedPMML = nextPMML;
132 
133         } else {
134 
135           Model currentModel = joinedPMML.getModels().get(0);
136           Segmentation segmentation;
137           if (currentModel instanceof MiningModel) {
138             // Already have a MiningModel, to which new models can be added in its Segmentation
139             segmentation = ((MiningModel) currentModel).getSegmentation();
140           } else {
141             // Lone tree model. Build a MiningModel around it, first, and swap it in
142             boolean classificationTask = currentModel.getFunctionName() == MiningFunctionType.CLASSIFICATION;
143             MiningModel miningModel = new MiningModel();
144             MultipleModelMethodType multipleModelMethodType = classificationTask ?
145                 MultipleModelMethodType.WEIGHTED_MAJORITY_VOTE :
146                 MultipleModelMethodType.WEIGHTED_AVERAGE;
147             segmentation = new Segmentation(multipleModelMethodType);
148             miningModel.setSegmentation(segmentation);
149             miningModel.setFunctionName(currentModel.getFunctionName());
150             MiningSchema cloneSchema = new MiningSchema();
151             for (MiningField field : currentModel.getMiningSchema().getMiningFields()) {
152               cloneSchema.getMiningFields().add(
153                   new MiningField(field.getName())
154                       .withOptype(field.getOptype()
155                       ).withImportance(field.getImportance()));
156             }
157             miningModel.setMiningSchema(cloneSchema);
158 
159             Segment segment = new Segment();
160             // This will get overriden below when renumbering happens
161             segment.setId("0");
162             segment.setPredicate(new True());
163             segment.setModel(currentModel);
164             // Don't actually know the weight, so use 1
165             segment.setWeight(1.0);
166             segmentation.getSegments().add(segment);
167             // Swap in new model
168             joinedPMML.getModels().clear();
169             joinedPMML.getModels().add(miningModel);
170           }
171 
172           Model nextModel = nextPMML.getModels().get(0);
173           if (nextModel instanceof MiningModel) {
174             segmentation.getSegments().addAll(((MiningModel) nextModel).getSegmentation().getSegments());
175           } else {
176             Segment segment = new Segment();
177             // This will get overriden below when renumbering happens
178             segment.setId("0");
179             segment.setPredicate(new True());
180             segment.setModel(nextModel);
181             // Don't actually know the weight, so use 1
182             segment.setWeight(1.0);
183             segmentation.getSegments().add(segment);
184           }
185 
186           updateDataDictionary(joinedPMML.getDataDictionary(), nextPMML.getDataDictionary());
187         }
188       }
189     }
190 
191     Preconditions.checkNotNull(joinedPMML, "No forests to join?");
192 
193     Model model = joinedPMML.getModels().get(0);
194 
195     if (model instanceof MiningModel) {
196 
197       // Renumber segments with distinct IDs
198       List<Segment> segments = ((MiningModel) model).getSegmentation().getSegments();
199       for (int treeID = 0; treeID < segments.size(); treeID++) {
200         segments.get(treeID).setId(Integer.toString(treeID));
201       }
202 
203       // Stitch together feature importances; only needed in an ensemble
204       for (MiningField field : model.getMiningSchema().getMiningFields()) {
205         String name = field.getName().getValue();
206         Mean importance = columnNameToMeanImportance.get(name);
207         if (importance == null) {
208           field.setImportance(null);
209         } else {
210           field.setImportance(importance.getResult());
211         }
212       }
213     }
214 
215     log.info("Writing combined model file");
216     File tempJoinedForestFile = File.createTempFile("model-", ".pmml.gz");
217     tempJoinedForestFile.deleteOnExit();
218     OutputStream out = IOUtils.buildGZIPOutputStream(new FileOutputStream(tempJoinedForestFile));
219     try {
220       JAXBUtil.marshalPMML(joinedPMML, new StreamResult(out));
221     } catch (JAXBException e) {
222       throw new IOException(e);
223     } finally {
224       out.close();
225     }
226 
227     log.info("Uploading combined model file");
228     store.upload(instanceGenerationPrefix + "model.pmml.gz", tempJoinedForestFile, false);
229     IOUtils.delete(tempJoinedForestFile);
230   }

232   private static void updateMeanImportances(Map<String,Mean> columnNameToMeanImportance,
233                                             List<MiningField> miningFields) {
234     for (MiningField field : miningFields) {
235       Double importance = field.getImportance();
236       if (importance != null) {
237         String fieldName = field.getName().getValue();
238         Mean mean = columnNameToMeanImportance.get(fieldName);
239         if (mean == null) {
240           mean = new Mean();
241           columnNameToMeanImportance.put(fieldName, mean);
242         }
243         mean.increment(importance);
244       }
245     }
246   }
247 
248   private static void updateDataDictionary(DataDictionary existingDict, DataDictionary newDict) {
249     for (TypeDefinitionField newField : newDict.getDataFields()) {
250       Collection<Value> newValues = newField.getValues();
251       if (newValues != null && !newValues.isEmpty()) {
252         for (TypeDefinitionField existingField : existingDict.getDataFields()) {
253           if (existingField.getName().equals(newField.getName())) {
254             Set<String> existingStrings = new HashSet<String>(existingField.getValues().size());
255             for (Value existingValue : existingField.getValues()) {
256               existingStrings.add(existingValue.getValue());
257             }
258             for (Value newValue : newValues) {
259               if (!existingStrings.contains(newValue.getValue())) {
260                 existingField.getValues().add(newValue);
261               }
262             }
263             break;
264           }
265         }
266       }
267     }
268   }
