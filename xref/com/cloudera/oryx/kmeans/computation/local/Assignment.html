16  package com.cloudera.oryx.kmeans.computation.local;
17  
18  import com.cloudera.oryx.common.math.NamedRealVector;
19  import com.cloudera.oryx.common.settings.ConfigUtils;
20  import com.cloudera.oryx.common.settings.InboundSettings;
21  import com.cloudera.oryx.kmeans.common.Distance;
22  import com.cloudera.oryx.kmeans.computation.evaluate.KMeansEvaluationData;
23  import com.google.common.base.Joiner;
24  import com.google.common.collect.ImmutableList;
25  import com.google.common.collect.Lists;
26  import com.typesafe.config.Config;
27  import org.apache.commons.math3.linear.RealVector;
28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;
30  
31  import java.util.List;
32  import java.util.concurrent.Callable;
33  
34  public class Assignment implements Callable<List<String>> {
35  
36    private static final Logger log = LoggerFactory.getLogger(Assignment.class);
37    private static final Joiner COMMA = Joiner.on(',');
38  
39    private final List<List<RealVector>> folds;
40    private final List<KMeansEvaluationData> clusters;
41  
42    public Assignment(List<List<RealVector>> folds, List<KMeansEvaluationData> clusters) {
43      this.folds = folds;
44      this.clusters = clusters;
45    }
46  
47    @Override
48    public List<String> call() {
49      Config config = ConfigUtils.getDefaultConfig();
50      List<String> ret = Lists.newArrayList();
51      if (doOutlierComputation(config)) {
52        InboundSettings inboundSettings = InboundSettings.create(config);
53        if (inboundSettings.getIdColumns().isEmpty()) {
54          log.error("Cluster assignments require that id-columns be configured for each vector");
55          return ImmutableList.of();
56        }
57        for (List<RealVector> fold : folds) {
58          for (RealVector vec : fold) {
59            NamedRealVector nvec = (NamedRealVector) vec;
60            for (KMeansEvaluationData data : clusters) {
61              Distance d = data.getBest().getDistance(nvec);
62              ret.add(COMMA.join(
63                  nvec.getName(), data.getK(), d.getClosestCenterId(), d.getSquaredDistance()));
64            }
65          }
66        }
67        return ret;
68      }
69      return ImmutableList.of();
70    }
71  
72    private static boolean doOutlierComputation(Config config) {
73      return config.hasPath("model.outliers") && config.getBoolean("model.outliers.compute");
74    }
75  }
